            {
                register int i,j;
              
              for(i = 0 ; i < array_size ; i++){
                for(j = i ; j < array_size ; j++){
                    if(x[i][j]!=x[j][i])

              } //end of for i
               
              return 1;

            void Most_Repeat(int A[], int len)
            {
                int MR_Value,      // Most Repeat Index
                    MR_Index,      // Most Repeat Value
                    Current_RValue;
            
                register int i, j;
            
                MR_Value = -1;
                for ( i = 0 ; i < len ; i ++){
            
                      Current_RValue = 0;
                      for(j = 0 ; j < len ; j++)
                          if (A[i] == A[j])
                              Current_RValue++;
            
                      if (Current_RValue > MR_Value) {
                          MR_Index = i;
                          MR_Value = Current_RValue;
                      }// end of if
            
                } // End of for ( i = 0 ...
            
                printf("\n\n > The Most Repeat : Item[%d] = %d \n > Repeat Value = %d\n\n",
                        MR_Index,A[MR_Index],MR_Value);
            
                return;
            }
            
            void Frequency_Repeat(int A[], int len)
            {
                int frequency[Base_random] = {};     
                register int i,j;
              
                
                for (i = 0 ; i < len ; i++)
                    ++frequency[A[i]];
                
                printf("\n\n%8s%10s%12s","Value","Repeat","Histogram");
                for(i = 0 ; i < Base_random ; i++){
                  
                  printf("\n%8d%10d    ",i,frequency[i]); 
                  
                    for(j=0 ; j < frequency[i] ; ++j)    
                        printf("%c",'*');
              }
                    
                return;
            }

              return;
            }
            
            void Bubble_Sort(int A[], int len)
            {
              register int i, j, temp;
            
              for (i = 1 ; i < len ; i++)
                  for (j = 0; j < len - i ; j++)
                       if (A[j] < A[j+1]){ //Descending Sort
                           temp = A[j];
                           A[j] = A[j+1];
                           A[j+1] = temp;
                         }
             return;
            }

  int main()
  {
      int A[array_length] = { 11, 12, 15, 17, 18, 19, 25, 28, 30, 35},
          B[array_length] = { 10, 17, 26, 27, 29, 30, 35, 40, 41, 50};
  
      int C[(2*array_length)];
  
      register int i;
  
      Merge_Array(A, B, C, array_length);
  
      for(i = 0 ; i < 2*array_length ; i++)
      {
          printf("\t C[%2d] = %3d", i, C[i]);
  
          if ((i+1) % 3 == 0)
              printf("\n");
      }
  
      //system("PAUSE");
      return 0;
  }
  
  void Merge_Array(int A[], int B[], int C[], int len)
  {
      register int i, j, k;
  
      i = j = k = 0;
  
      while( i  < len && j < len ){
          if ( A[i] > B[j] )
              C[k++] = B[j++];
          else
              C[k++] = A[i++];
      }
  
      while (i < len){
             C[k++] = A[i++];
      }
  
      while (j < len){
             C[k++] = B[j++];
      }
  
      return;
  }

          #include <stdio.h>
          #include <stdlib.h>
            
            unsigned long long int Fact (int n)
            { 
              unsigned long long int F = 1;
              while(n>1){
                  F*=n;
                  n--;
              }
              
              return F;
            }
            
            unsigned int Combination(int C,int r)
            {
              return(Fact(C)/(Fact(r)*Fact(C-r)));
            }
            
            int main()
            {
              int num;
              
              printf("Enter a number: ");
              scanf("%d",&num);
              
              for (size_t c = 0; c<=num ; c++){
                
                for (size_t r = 0 ; r<=c ; r++){
                     printf("%5u",Combination(c,r))	;
                }//end of for r (Internal Loop)
                
                printf("\n");
                
              }//end of for c (External Loop)
              
              system("PAUSE");
              return 0;
            }
